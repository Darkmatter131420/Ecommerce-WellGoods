商品：

```MongoDB

db.products.insertOne(
{
  "_id": 1, // 自增id
  "name": "商品名称",         // 必填
  "description": "商品描述",   // 默认为null
  "price": 100.00, // 高精度十进制类型，必填，大于或等于0
  "category": ["分类1","分类2"],       // 必填
  "merchant_id": 1,         // 商家id，必填，与用户集合关联
  "picture": ["图片1","图片2"], // 商品图片，必填
  "status": 0,
  "create_time": new Date, // 创建时间
  "update_time": new Date // 更新时间
})
```
